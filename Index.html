puntos = 0

def menu():

    print("Si te equivocas, se restará 1 punto.")
    print("¡Tranquil@, tendrás pequeñas pistas!")

    opcion = input("Elige una opción:")
    if opcion == '1':

def empezar():
    global puntos
    n1 = randint(1, 100)
    n2 = randint(1, 100)
    suma = n1 + n2
    
    print("A ver si puedes con esto:")
    rta = int(input(f"¿Cuánto es {n1} + {n2}?: "))
    
    if rta == suma:
        print("¡Correcto! ¡Has acertado!")
        puntos += 1
    else:
        print("¡Te has equivocado!")
        puntos -= 1
    
    print(f"Puntos actuales: {puntos}")
    menu()
